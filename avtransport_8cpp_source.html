59AVTransport::AVTransport(const std::string& serviceHost, unsigned servicePort, SubscriptionPoolPtr& subscriptionPool, void* CBHandle, EventCB eventCB)
60: Service(serviceHost, servicePort)
61, m_subscriptionPool(subscriptionPool)
62, m_subscription()
63, m_CBHandle(CBHandle)
64, m_eventCB(eventCB)
65, m_msgCount(0)
66, m_property(AVTProperty())
67, m_lastPositionInfo(new AVTransportLastInfo())
68{
69 unsigned subId = m_subscriptionPool->GetEventHandler().CreateSubscription(this);
70 m_subscriptionPool->GetEventHandler().SubscribeForEvent(subId, EVENT_UPNP_PROPCHANGE);
71 m_subscription = m_subscriptionPool->SubscribeEvent(serviceHost, servicePort, EventURL);
72 m_subscription.Start();
73}

95bool AVTransport::GetPositionInfo(ElementList& vars)
96{
97 Locked<AVTransportLastInfo*>::pointer p = m_lastPositionInfo.Get();
98 if ((*p)->expiry.TimeLeft() > 0)
99 {
100 vars = (*p)->vars;
101 return true;
102 }
103 ElementList args;
104 args.push_back(ElementPtr(new Element("InstanceID", "0")));
105 vars = Request("GetPositionInfo", args);
106 if (!vars.empty() && vars[0]->compare("GetPositionInfoResponse") == 0)
107 {
108 (*p)->vars = vars;
109 (*p)->expiry.Set(1000); // expire in 1 second
110 return true;
111 }
112 return false;

157 }

292unsigned AVTransport::AddMultipleURIsToQueue(const std::vector<std::string>& uris, const std::vector<std::string>& metadatas)

294 ElementList args;
295 Element* elem;
296 args.push_back(ElementPtr(new Element("InstanceID", "0")));

298 args.push_back(ElementPtr(new Element("NumberOfURIs", std::to_string(uris.size()))));
299 elem = new Element("EnqueuedURIs", "");
300 for (std::vector<std::string>::const_iterator it = uris.begin(); it != uris.end(); ++it)

304 elem->append(*it);
305 }
306 args.push_back(ElementPtr(elem));
307 elem = new Element("EnqueuedURIsMetaData", "");
308 for (std::vector<std::string>::const_iterator it = metadatas.begin(); it != metadatas.end(); ++it)
309 {
310 if (it != metadatas.begin())
311 elem->append(" ");
312 elem->append(*it);
313 }
314 args.push_back(ElementPtr(elem));
315 args.push_back(ElementPtr(new Element("ContainerURI", "")));
316 args.push_back(ElementPtr(new Element("ContainerMetadata", "")));
317 args.push_back(ElementPtr(new Element("DesiredFirstTrackNumberEnqueued", "0")));
318 args.push_back(ElementPtr(new Element("EnqueueAsNext", "0")));
319 ElementList vars = Request("AddMultipleURIsToQueue", args);

481void AVTransport::HandleEventMessage(EventMessagePtr msg)
482{
483 if (!msg)
484 return;
485 if (msg->event == EVENT_UPNP_PROPCHANGE)
486 {
487 if (m_subscription.GetSID() == msg->subject[0] && msg->subject[2] == "AVT")
488 {
489 {
490 // BEGIN CRITICAL SECTION
491 Locked<AVTProperty>::pointer prop = m_property.Get();
492
493 DBG(DBG_DEBUG, "%s: %s SEQ=%s %s\n", __FUNCTION__, msg->subject[0].c_str(), msg->subject[1].c_str(), msg->subject[2].c_str());
494
495 // check for higher sequence
496 uint32_t seq = 0;
497 string_to_uint32(msg->subject[1].c_str(), &seq);
498 if (msg->subject[0] != prop->EventSID)
499 {
500 prop->EventSID = msg->subject[0];
501 }
502 else if (seq < prop->EventSEQ)
503 {
504 DBG(DBG_DEBUG, "%s: %s SEQ=%u , discarding %u\n", __FUNCTION__, prop->EventSID.c_str(), prop->EventSEQ, seq);
505 return;
506 }
507 // tracking serial of the event
508 prop->EventSEQ = seq;
509
510 std::vector<std::string>::const_iterator it = msg->subject.begin();
511 while (it != msg->subject.end())
512 {
513 uint32_t num;
514 if (*it == "TransportState")
515 prop->TransportState.assign(*++it);
516 else if (*it == "CurrentPlayMode")
517 prop->CurrentPlayMode.assign(*++it);
518 else if (*it == "CurrentCrossfadeMode")
519 prop->CurrentCrossfadeMode.assign(*++it);
520 else if (*it == "NumberOfTracks")
521 {
522 string_to_uint32((*++it).c_str(), &num);
523 prop->NumberOfTracks = (unsigned)num;
524 }
525 else if (*it == "CurrentTrack")
526 {
527 string_to_uint32((*++it).c_str(), &num);
528 prop->CurrentTrack = (unsigned)num;
529 }
530 else if (*it == "CurrentSection")
531 {
532 string_to_uint32((*++it).c_str(), &num);
533 prop->CurrentSection = (unsigned)num;
534 }
535 else if (*it == "CurrentTrackURI")
536 prop->CurrentTrackURI.assign(*++it);
537 else if (*it == "CurrentTrackDuration")
538 prop->CurrentTrackDuration.assign(*++it);
539 else if (*it == "CurrentTrackMetaData")
540 {
541 DIDLParser didl((*++it).c_str());
542 if (didl.IsValid() && !didl.GetItems().empty())
543 prop->CurrentTrackMetaData = didl.GetItems()[0];
544 else
545 prop->CurrentTrackMetaData.reset(new DigitalItem(DigitalItem::Type_unknown));
546 }
547 else if (*it == "r:NextTrackURI")
548 prop->r_NextTrackURI.assign(*++it);
549 else if (*it == "r:NextTrackMetaData")
550 {
551 DIDLParser didl((*++it).c_str());
552 if (didl.IsValid() && !didl.GetItems().empty())
553 prop->r_NextTrackMetaData = didl.GetItems()[0];
554 else
555 prop->r_NextTrackMetaData.reset(new DigitalItem(DigitalItem::Type_unknown));
556 }
557 else if (*it == "r:EnqueuedTransportURI")
558 prop->r_EnqueuedTransportURI.assign(*++it);
559 else if (*it == "r:EnqueuedTransportURIMetaData")
560 {
561 DIDLParser didl((*++it).c_str());
562 if (didl.IsValid() && !didl.GetItems().empty())
563 prop->r_EnqueuedTransportURIMetaData = didl.GetItems()[0];
564 else
565 prop->r_EnqueuedTransportURIMetaData.reset(new DigitalItem(DigitalItem::Type_unknown));
566 }
567 else if (*it == "PlaybackStorageMedium")
568 prop->PlaybackStorageMedium.assign(*++it);
569 else if (*it == "AVTransportURI")
570 prop->AVTransportURI.assign(*++it);
571 else if (*it == "AVTransportURIMetaData")
572 {
573 DIDLParser didl((*++it).c_str());
574 if (didl.IsValid() && !didl.GetItems().empty())
575 prop->AVTransportURIMetaData = didl.GetItems()[0];
576 else
577 prop->AVTransportURIMetaData.reset(new DigitalItem(DigitalItem::Type_unknown));
578 }
579 else if (*it == "NextAVTransportURI")
580 prop->NextAVTransportURI.assign(*++it);
581 else if (*it == "NextAVTransportURIMetaData")
582 prop->NextAVTransportURIMetaData.assign(*++it);
583 else if (*it == "CurrentTransportActions")
584 prop->CurrentTransportActions.assign(*++it);
585 else if (*it == "r:CurrentValidPlayModes")
586 prop->r_CurrentValidPlayModes.assign(*++it);
587 else if (*it == "r:MuseSessions")
588 prop->r_MuseSessions.assign(*++it);
589 else if (*it == "TransportStatus")
590 prop->TransportStatus.assign(*++it);
591 else if (*it == "r:SleepTimerGeneration")
592 prop->r_SleepTimerGeneration.assign(*++it);
593 else if (*it == "r:AlarmRunning")
594 prop->r_AlarmRunning.assign(*++it);
595 else if (*it == "r:AlarmIDRunning")
596 prop->r_AlarmIDRunning.assign(*++it);
597 else if (*it == "r:AlarmLoggedStartTime")
598 prop->r_AlarmLoggedStartTime.assign(*++it);
599 else if (*it == "r:AlarmState")
600 prop->r_AlarmState.assign(*++it);
601 else if (*it == "r:SnoozeRunning")
602 prop->r_SnoozeRunning.assign(*++it);
603 else if (*it == "r:RestartPending")
604 prop->r_RestartPending.assign(*++it);
605 else if (*it == "PossiblePlaybackStorageMedia")
606 prop->PossiblePlaybackStorageMedia.assign(*++it);
607
608 ++it;
609 }
610 // END CRITICAL SECTION
611 }
612 // Signal
613 ++m_msgCount;
614 if (m_eventCB)
615 m_eventCB(m_CBHandle);
616 }
617 }
