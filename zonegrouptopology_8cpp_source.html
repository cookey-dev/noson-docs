34inline uint_fast32_t __hashvalue(uint_fast32_t maxsize, const char *value)
35{
36 uint_fast32_t h = 0, g;
37
38 while (*value)
39 {
40 h = (h << 4) + *value++;
41 if ((g = h & 0xF0000000L))
42 {
43 h ^= g >> 24;
44 }
45 h &= ~g;
46 }
47
48 return h % maxsize;
49}

101void ZoneGroupTopology::HandleEventMessage(EventMessagePtr msg)
102{
103 if (!msg)
104 return;
105 if (msg->event == EVENT_UPNP_PROPCHANGE)
106 {
107 if (m_subscription.GetSID() == msg->subject[0] && msg->subject[2] == "PROPERTY")
108 {
109 DBG(DBG_DEBUG, "%s: %s SEQ=%s %s\n", __FUNCTION__, msg->subject[0].c_str(), msg->subject[1].c_str(), msg->subject[2].c_str());
110
111 // check for higher sequence
112 uint32_t seq = 0;
113 string_to_uint32(msg->subject[1].c_str(), &seq);
114 if (msg->subject[0] != m_eventSID)
115 {
116 m_eventSID = msg->subject[0];
117 }
118 else if (seq < m_eventSEQ)
119 {
120 DBG(DBG_DEBUG, "%s: %s SEQ=%u , discarding %u\n", __FUNCTION__, m_eventSID.c_str(), m_eventSEQ, seq);
121 return;
122 }
123 // tracking serial of the event
124 m_eventSEQ = seq;
125
126 std::vector<std::string>::const_iterator it = msg->subject.begin();
127 unsigned _oldKey = m_topologyKey;
128 while (it != msg->subject.end())
129 {
130 if (*it == "ZoneGroupState")
131 {
132 // BEGIN CRITICAL SECTION
133 ParseZoneGroupState(*++it);
134 // END CRITICAL SECTION
135 break;
136 }
137 ++it;
138 }
139 // Event is signaled only on first or any change
140 if (m_msgCount && _oldKey == m_topologyKey)
141 return;
142 // Signal
143 ++m_msgCount;
144 if (m_eventCB)
145 m_eventCB(m_CBHandle);
146 }
147 }
148}

150bool ZoneGroupTopology::ParseZoneGroupState(const std::string& xml)
151{
152 tinyxml2::XMLDocument rootdoc;
153 // Parse xml content
154 if (rootdoc.Parse(xml.c_str(), xml.size()) != tinyxml2::XML_SUCCESS)
155 {
156 DBG(DBG_ERROR, "%s: parse xml failed\n", __FUNCTION__);
157 return false;
158 }
159
160 const tinyxml2::XMLElement* elem; // an element
161 if ((elem = rootdoc.RootElement()))
162 {
163 // since API version 10.2 (build 50163230): the root element is 'ZoneGroupState'
164 if (XMLNS::NameEqual(elem->Name(), "ZoneGroupState"))
165 elem = elem->FirstChildElement();
166 }
167 else
168 {
169 DBG(DBG_ERROR, "%s: invalid or not supported content\n", __FUNCTION__);
170 tinyxml2::XMLPrinter out;
171 rootdoc.Accept(&out);
172 DBG(DBG_ERROR, "%s\n", out.CStr());
173 return false;
174 }
175
176 Locked<ZoneList>::pointer zones = m_zones.Get();
177 Locked<ZonePlayerList>::pointer zonePlayers = m_zonePlayers.Get();
178 zones->clear();
179 zonePlayers->clear();
180
181 while (elem)
182 {
183 // VanishedDevices: Added in API version 10.2 (build 50163230)
184 // ZoneGroups
185 if (XMLNS::NameEqual(elem->Name(), "ZoneGroups"))
186 {
187 const tinyxml2::XMLElement* gelem = elem->FirstChildElement();
188 while (gelem)
189 {
190 const tinyxml2::XMLAttribute* attr = gelem->FirstAttribute();
191 Element zoneGroup(gelem->Name());
192 while (attr)
193 {
194 zoneGroup.SetAttribut(attr->Name(), attr->Value());
195 attr = attr->Next();
196 }
197 ZonePtr zone(new Zone(zoneGroup.GetAttribut("ID")));
198 const std::string& cuuid = zoneGroup.GetAttribut("Coordinator");
199 DBG(DBG_INFO, "%s: new group '%s' with coordinator '%s'\n", __FUNCTION__, zone->GetGroup().c_str(), cuuid.c_str());
200 // browse childs
201 const tinyxml2::XMLElement* child = gelem->FirstChildElement();
202 while (child)
203 {
204 if (XMLNS::NameEqual(child->Name(), "ZoneGroupMember"))
205 {
206 const tinyxml2::XMLAttribute* cattr = child->FirstAttribute();
207 Element zoneGroupMember(child->Name());
208 while (cattr)
209 {
210 zoneGroupMember.SetAttribut(cattr->Name(), cattr->Value());
211 cattr = cattr->Next();
212 }
213 if (zoneGroupMember.GetAttribut("Invisible") == "1")
214 {
215 const std::string& mname = zoneGroupMember.GetAttribut("ZoneName");
216 const std::string& muuid = zoneGroupMember.GetAttribut("UUID");
217 DBG(DBG_INFO, "%s: discard invisible group member '%s' (%s)\n", __FUNCTION__, muuid.c_str(), mname.c_str());
218 }
219 else
220 {
221 ZonePlayerPtr zp(new ZonePlayer(zoneGroupMember.GetAttribut("ZoneName")));
222 const std::string& muuid = zoneGroupMember.GetAttribut("UUID");
223 zp->SetAttribut(ZP_UUID, muuid);
224 if (muuid == cuuid)
225 zp->SetAttribut(ZP_COORDINATOR, "true");
226 else
227 zp->SetAttribut(ZP_COORDINATOR, "false");
228 zp->SetAttribut(ZP_LOCATION, zoneGroupMember.GetAttribut("Location"));
229 zp->SetAttribut(ZP_ICON, zoneGroupMember.GetAttribut("Icon"));
230 zp->SetAttribut(ZP_VERSION, zoneGroupMember.GetAttribut("SoftwareVersion"));
231 zp->SetAttribut(ZP_MCVERSION, zoneGroupMember.GetAttribut("MinCompatibleVersion"));
232 zp->SetAttribut(ZP_LCVERSION, zoneGroupMember.GetAttribut("LegacyCompatibleVersion"));
233 DBG(DBG_INFO, "%s: new group member '%s' (%s)\n", __FUNCTION__, muuid.c_str(), zp->c_str());
234 zonePlayers->insert(std::make_pair(*zp, zp));
235 zone->push_back(zp);
236 }
237 }
238 child = child->NextSiblingElement(NULL);
239 }
240 if (!zone->empty())
241 {
242 zone->Revamp();
243 zones->insert(std::make_pair(zone->GetGroup(), zone));
244 }
245 gelem = gelem->NextSiblingElement(NULL);
246 }
247 }
248 elem = elem->NextSiblingElement(NULL);
249 }
250 // compute a key for this state
251 std::string keyStr;
252 keyStr.reserve(zones->size() << 5);
253 for (ZoneList::const_iterator it = zones->begin(); it != zones->end(); ++it)
254 keyStr.append(it->first);
255 m_topologyKey = __hashvalue(0xFFFFFFFF, keyStr.c_str());
256 DBG(DBG_INFO, "%s: topology key %u\n", __FUNCTION__, m_topologyKey);
257 return (!zones->empty());
258}
