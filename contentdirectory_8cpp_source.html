70bool ContentDirectory::Browse(const std::string& objectId, unsigned index, unsigned count, ElementList &vars)
71{
72 ElementList args;
73 args.push_back(ElementPtr(new Element("ObjectID", objectId)));
74 args.push_back(ElementPtr(new Element("BrowseFlag", "BrowseDirectChildren")));
75 args.push_back(ElementPtr(new Element("Filter", "*")));
76 args.push_back(ElementPtr(new Element("StartingIndex", std::to_string(index))));
77 args.push_back(ElementPtr(new Element("RequestedCount", std::to_string(count))));
78 args.push_back(ElementPtr(new Element("SortCriteria", "")));
79 vars = Request("Browse", args);
80 if (!vars.empty() && vars[0]->compare("BrowseResponse") == 0)
81 return true;
82 return false;

119void ContentDirectory::HandleEventMessage(EventMessagePtr msg)
120{
121 if (!msg)
122 return;
123 if (msg->event == EVENT_UPNP_PROPCHANGE)
124 {
125 if (m_subscription.GetSID() == msg->subject[0] && msg->subject[2] == "PROPERTY")
126 {
127 {
128 // BEGIN CRITICAL SECTION
129 Locked<ContentProperty>::pointer prop = m_property.Get();
130
131 DBG(DBG_DEBUG, "%s: %s SEQ=%s %s\n", __FUNCTION__, msg->subject[0].c_str(), msg->subject[1].c_str(), msg->subject[2].c_str());
132
133 // check for higher sequence
134 uint32_t seq = 0;
135 string_to_uint32(msg->subject[1].c_str(), &seq);
136 if (msg->subject[0] != prop->EventSID)
137 {
138 prop->EventSID = msg->subject[0];
139 }
140 else if (seq < prop->EventSEQ)
141 {
142 DBG(DBG_DEBUG, "%s: %s SEQ=%u , discarding %u\n", __FUNCTION__, prop->EventSID.c_str(), prop->EventSEQ, seq);
143 return;
144 }
145 // tracking serial of the event
146 prop->EventSEQ = seq;
147
148 std::vector<std::string>::const_iterator it = msg->subject.begin();
149 while (it != msg->subject.end())
150 {
151 if (*it == "SystemUpdateID")
152 prop->SystemUpdateID.assign(*++it);
153 else if (*it == "ContainerUpdateIDs")
154 {
155 prop->ContainerUpdateIDs.clear();
156 std::vector<std::string> tokens;
157 tokenize((*++it).c_str(), ",", tokens);
158 std::vector<std::string>::const_iterator itt = tokens.begin();
159 while (itt != tokens.end())
160 {
161 const std::string& str = *itt;
162 if (++itt != tokens.end())
163 {
164 uint32_t num;
165 if (string_to_uint32(itt->c_str(), &num) == 0)
166 prop->ContainerUpdateIDs.emplace_back(str, num);
167 }
168 }
169 }
170 else if (*it == "UserRadioUpdateID")
171 prop->UserRadioUpdateID.assign(*++it);
172 else if (*it == "SavedQueuesUpdateID")
173 prop->SavedQueuesUpdateID.assign(*++it);
174 else if (*it == "ShareListUpdateID")
175 prop->ShareListUpdateID.assign(*++it);
176 else if (*it == "RecentlyPlayedUpdateID")
177 prop->RecentlyPlayedUpdateID.assign(*++it);
178 else if (*it == "RadioFavoritesUpdateID")
179 prop->RadioFavoritesUpdateID.assign(*++it);
180 else if (*it == "RadioLocationUpdateID")
181 prop->RadioLocationUpdateID.assign(*++it);
182 else if (*it == "FavoritesUpdateID")
183 prop->FavoritesUpdateID.assign(*++it);
184 else if (*it == "FavoritePresetsUpdateID")
185 prop->FavoritePresetsUpdateID.assign(*++it);
186 else if (*it == "ShareIndexInProgress")
187 {
188 int32_t num;
189 string_to_int32((++it)->c_str(), &num);
190 prop->ShareIndexInProgress = (num != 0);
191 }
192
193 ++it;
194 }
195 // END CRITICAL SECTION
196 }
197 // Signal
198 if (m_eventCB)
199 m_eventCB(m_CBHandle);
200 }
201 }
202}

248unsigned ContentChunk::summarize(const ElementList& vars)
249{
250 uint32_t updateID = 0;
251 if (string_to_uint32(vars.GetValue("UpdateID").c_str(), &updateID) == 0)
252 m_lastUpdateID = updateID; // set update ID for this chunk of data
253 uint32_t totalcount = 0;
254 if (string_to_uint32(vars.GetValue("TotalMatches").c_str(), &totalcount) == 0)
255 m_totalCount = totalcount; // reset total count
256 uint32_t cnt = 0;
257 string_to_uint32(vars.GetValue("NumberReturned").c_str(), &cnt);
258 return (unsigned)cnt;
259}

317bool ContentList::BrowseContent(unsigned startingIndex, unsigned count, List::iterator position)
318{
319 DBG(DBG_PROTO, "%s: browse %u from %u\n", __FUNCTION__, count, startingIndex);
320 ElementList vars;
321 ElementList::const_iterator it;
322 if ((m_succeeded = m_service.Browse(m_root, startingIndex, count, vars)) && (it = vars.FindKey("Result")) != vars.end())
323 {
324 unsigned cnt = summarize(vars);
325 // peer could return a valid result on out of range
326 if (startingIndex < m_totalCount)
327 {
328 DIDLParser didl((*it)->c_str(), cnt);
329 if (didl.IsValid())
330 {
331 m_list.insert(position, didl.GetItems().begin(), didl.GetItems().end());
332 m_browsedCount += didl.GetItems().size();
333 DBG(DBG_PROTO, "%s: count %u\n", __FUNCTION__, didl.GetItems().size());
334 return true;
335 }
336 }
337 }
338 return false;
339}

364bool ContentBrowser::Browse(unsigned index, unsigned count)
365{
366 if (index >= m_totalCount)
367 {
368 m_table.clear();
369 m_startingIndex = m_totalCount;
370 return false;
371 }
372
373 unsigned size = (unsigned) m_table.size();
374 if (m_totalCount < index + count)
375 count = m_totalCount - index;
376
377 if (index == m_startingIndex)
378 {
379 if (count == size)
380 return true;
381 if (count < size)
382 {
383 m_table.resize(count);
384 }
385 return BrowseContent(m_startingIndex + size, count - size, m_table.end());
386 }
387
388 if (index > m_startingIndex && index + count <= m_startingIndex + size)
389 {
390 Table tmp;
391 tmp.insert(tmp.begin(), m_table.begin() + index - m_startingIndex, m_table.begin() + index - m_startingIndex + count);
392 m_table = tmp;
393 m_startingIndex = index;
394 return true;
395 }
396
397 m_table.clear();
398 m_startingIndex = index;
399 return BrowseContent(m_startingIndex, count, m_table.begin());
400}
